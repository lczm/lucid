704 // bool : true if collided, false if not collided
705 // float : length of the intersection from origin if it is intersected
706 std::tuple<bool, float> RenderSystem::RayBoundingBoxCollisionCheck(glm::vec3 origin, glm::vec3 ray,
707  BoundingBox boundingBox)
708 {
709  float length;
710  glm::vec3 dirfrac = 1.0f / ray;
711 
712  float t1 = (boundingBox.minX - origin.x) * dirfrac.x;
713  float t2 = (boundingBox.maxX - origin.x) * dirfrac.x;
714 
715  float t3 = (boundingBox.minY - origin.y) * dirfrac.y;
716  float t4 = (boundingBox.maxY - origin.y) * dirfrac.y;
717 
718  float t5 = (boundingBox.minZ - origin.z) * dirfrac.z;
719  float t6 = (boundingBox.maxZ - origin.z) * dirfrac.z;
720 
721  float tmin = glm::max(glm::max(glm::min(t1, t2), glm::min(t3, t4)), glm::min(t5, t6));
722  float tmax = glm::min(glm::min(glm::max(t1, t2), glm::max(t3, t4)), glm::max(t5, t6));
723 
724  // AABB is behind
725  if (tmax < 0)
726  {
727  return std::tuple(false, 0.0f);
728  }
729 
730  // Does not intersect
731  if (tmin > tmax)
732  {
733  return std::tuple(false, 0.0f);
734  }
735 
736  length = tmin;
737  return std::tuple(true, length);
738 }

740 void RenderSystem::InitSphereVerticesIndices(SphereVerticesIndices& sphereVerticesIndices)
741 {
742  // vertex positions
743  float x, y, z, xy;
744 
745  // vertex normals
746  // float nx, ny, nz;
747  // float lengthInverse = 1.0f / radius;
748  // float s, t;
749 
750  auto radius = sphereVerticesIndices.radius;
751  auto sectors = sphereVerticesIndices.sectors;
752  auto stacks = sphereVerticesIndices.stacks;
753 
754  float sectorStep = 2 * PI / sectors;
755  float stackStep = PI / stacks;
756  float sectorAngle, stackAngle;
757 
758  for (int i = 0; i <= stacks; i++)
759  {
760  stackAngle = PI / 2 - i * stackStep;
761 
762  xy = radius * cosf(stackAngle);
763  z = radius * sinf(stackAngle);
764 
765  // add (sectorCount+1) vertices per stack
766  // the first and last vertices have same position and normal, but different texCoord
767  for (int j = 0; j <= sectors; j++)
768  {
769  sectorAngle = j * sectorStep;
770 
771  // vertex position (x, y, z)
772  x = xy * cosf(sectorAngle);
773  y = xy * sinf(sectorAngle);
774  sphereVerticesIndices.vertices.push_back(x);
775  sphereVerticesIndices.vertices.push_back(y);
776  sphereVerticesIndices.vertices.push_back(z);
777 
778  // Note : Normals are not used at the current time
779  // normalized vertex normal (nx, ny, nz)
780  // nx = x * lengthInverse;
781  // ny = y * lengthInverse;
782  // nz = z * lengthInverse;
783  // normals.push_back(nx);
784  // normals.push_back(ny);
785  // normals.push_back(nz);
786 
787  // Note : TexCoords are also not used at the current time
788  // simplicity
789  // vertex tex coord (s, t) range between [0, 1]
790  // s = (float)j / sectors;
791  // t = (float)i / stacks;
792 
793  // texCoords.push_back(s);
794  // texCoords.push_back(t);
795  }
796  }
797 
798  int k1, k2;
799  for (int i = 0; i < stacks; ++i)
800  {
801  k1 = i * (sectors + 1); // beginning of current stack
802  k2 = k1 + sectors + 1; // beginning of next stack
803 
804  for (int j = 0; j < sectors; ++j, ++k1, ++k2)
805  {
806  // 2 triangles per sector excluding first and last stacks
807  // k1 => k2 => k1+1
808  if (i != 0)
809  {
810  sphereVerticesIndices.indices.push_back(k1);
811  sphereVerticesIndices.indices.push_back(k2);
812  sphereVerticesIndices.indices.push_back(k1 + 1);
813  }
814 
815  // k1+1 => k2 => k2+1
816  if (i != (stacks - 1))
817  {
818  sphereVerticesIndices.indices.push_back(k1 + 1);
819  sphereVerticesIndices.indices.push_back(k2);
820  sphereVerticesIndices.indices.push_back(k2 + 1);
821  }
822  }
823  }
824 }
