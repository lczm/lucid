134  // Update the internal fps counter before resetting it back to 0
135  DevDebug& devDebug = registry->GetEditorResource<DevDebug>();
136  devDebug.fps = frameCount;
137 
138  secondDt -= 1.0f;
139  frameCount = 0;
140  }
141 
142  // Get the current game state
143  GameEngineState& gameEngineState = registry->GetEditorResource<GameEngineState>();
144 
145 #if DEBUG
146  // Only if the game state is 'playing', then update the systems.
147  if (gameEngineState.gameState == GameState::PLAYING)
148  {
149  registry->UpdateSystems(dt, input);
150  }
151  else if (gameEngineState.gameState == GameState::PAUSED)
152  {
153  // If it is paused, draw the ui, and run the render system only.
154  // Note that this systems update has to be ran in the order of what the original systems
155  // are ran in
156  registry->UpdateSystem(dt, input, "ui");
157  registry->UpdateSystem(dt, input, "render");
158  }
159 #endif

161 #if RELEASE
