444 static void PanCamera(float dt, Camera* camera, Transform* transform, float offsetX, float offsetY)
445 {
446  const float damp = 0.00005f;
447 
448  offsetX *= CAMERA_SENSITIVITY * dt;
449  // This is always negative so that 'moving' the camera downwards feels natural
450  // As when it is 'right -clicked', it is in the opposite direction
451  offsetY *= -(CAMERA_SENSITIVITY * dt);
452 
453  camera->yaw += offsetX;
454  camera->pitch += offsetY;
455 
456  if (camera->pitch > twoPi)
457  {
458  camera->pitch -= twoPi;
459  }
460  else if (camera->pitch < -twoPi)
461  {
462  camera->pitch += twoPi;
463  }
464 
465  if (camera->yaw > twoPi)
466  {
467  camera->yaw -= twoPi;
468  }
469  else if (camera->yaw < -twoPi)
470  {
471  camera->yaw += twoPi;
472  }
473 
474  if (camera->roll > twoPi)
475  {
476  camera->roll -= twoPi;
477  }
478  else if (camera->roll < -twoPi)
479  {
480  camera->roll += twoPi;
481  }
482 
483  if (camera->pitch != 0.0f)
484  {
485  transform->rotation = RotateQuatX(transform->rotation, camera->pitch);
486  }
487 
488  if (camera->yaw != 0.0f)
489  {
490  transform->rotation = RotateQuatY(transform->rotation, camera->yaw);
491  }
492 
493  if (camera->roll != 0.0f)
494  {
495  transform->rotation = RotateQuatZ(transform->rotation, camera->roll);
496  }
497 
498  // camera->pitch *= damp;
499  // camera->yaw *= damp;
500  // camera->roll *= damp;
501 
502  camera->pitch *= 0;
503  camera->yaw *= 0;
504  camera->roll *= 0;
505 }
